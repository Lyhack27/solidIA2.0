            constructor(x, y, directionX, directionY, size, color) {
                this.x = x;
                this.y = y;
                this.directionX = directionX;
                this.directionY = directionY;
                this.size = size;
                this.color = color;
            }

            // Method to draw particle
            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2, false);
                ctx.fillStyle = this.color;
                ctx.fill();
            }

            // Method to update particle position and connect to others
            update() {
                // Bounce off edges
                if (this.x + this.size > canvas.width || this.x - this.size < 0) {
                    this.directionX = -this.directionX;
                }
                if (this.y + this.size > canvas.height || this.y - this.size < 0) {
                    this.directionY = -this.directionY;
                }

                this.x += this.directionX;
                this.y += this.directionY;

                // Mouse interaction - Adjusted for smoother touch
                let dx = mouse.x - this.x;
                let dy = mouse.y - this.y;
                let distance = Math.sqrt(dx * dx + dy * dy);

                // If within mouse radius, gently push particles away
                if (distance < mouse.radius + this.size) {
                    const pushStrength = 2; // Further reduced for even smoother interaction
                    if (mouse.x < this.x && this.x < canvas.width - this.size * pushStrength) {
                        this.x += pushStrength;
                    }
                    if (mouse.x > this.x && this.x > this.size * pushStrength) {
                        this.x -= pushStrength;
                    }
                    if (mouse.y < this.y && this.y < canvas.height - this.size * pushStrength) {
                        this.y += pushStrength;
                    }
                    if (mouse.y > this.y && this.y > this.size * pushStrength) {
                        this.y -= pushStrength;
                    }
                }

                this.draw();
            }
        }

        // Create particles
        function init() {
            particles = [];
            for (let i = 0; i < numParticles; i++) {
                let size = (Math.random() * 3) + 1; // Particle size between 1 and 4
                let x = (Math.random() * ((canvas.width - size * 2) - (size * 2)) + size * 2);
                let y = (Math.random() * ((canvas.height - size * 2) - (size * 2)) + size * 2);
                let directionX = (Math.random() * 0.5) - 0.25; // Speed between -0.25 and 0.25
                let directionY = (Math.random() * 0.5) - 0.25;
                let color = '#888888'; // Greyish color for nodes
                particles.push(new Particle(x, y, directionX, directionY, size, color));
            }
        }

        // Connect particles with lines
        function connect() {
            let opacityValue = 1;
            for (let a = 0; a < particles.length; a++) {
                for (let b = a; b < particles.length; b++) {
                    let distance = ((particles[a].x - particles[b].x) * (particles[a].x - particles[b].x)) +
                                   ((particles[a].y - particles[b].y) * (particles[a].y - particles[b].y));
                    if (distance < (maxDistance * maxDistance)) {
                        opacityValue = 1 - (distance / (maxDistance * maxDistance));
                        ctx.strokeStyle = `rgba(0, 150, 255, ${opacityValue})`; // Blue lines with varying opacity
                        ctx.lineWidth = 1;
                        ctx.beginPath();
                        ctx.moveTo(particles[a].x, particles[a].y);
                        ctx.lineTo(particles[b].x, particles[b].y);
                        ctx.stroke();
                    }
                }
            }
        }

        // Animation loop
        function animate() {
            requestAnimationFrame(animate);
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            for (let i = 0; i < particles.length; i++) {
                particles[i].update();
            }
            connect();
        }

        // Gemini API Integration and Contact Form Logic
        document.addEventListener('DOMContentLoaded', () => {
            const businessProblemInput = document.getElementById('businessProblemInput');
            const generateSolutionBtn = document.getElementById('generateSolutionBtn');
            const solutionOutput = document.getElementById('solutionOutput');
            const solutionText = document.getElementById('solutionText');
            const loadingIndicator = document.getElementById('loadingIndicator');
            const contactFormContainer = document.getElementById('contactFormContainer');
            const contactForm = document.getElementById('contactForm');
            const contactConfirmation = document.getElementById('contactConfirmation');

            generateSolutionBtn.addEventListener('click', async () => {
                const prompt = businessProblemInput.value.trim();
                if (!prompt) {
                    solutionOutput.classList.remove('hidden');
                    solutionText.textContent = 'Please enter a business problem to get a solution idea.';
                    contactFormContainer.classList.add('hidden'); // Hide form if no prompt
                    return;
                }

                // Show loading indicator, clear previous output and hide form
                solutionOutput.classList.remove('hidden');
                solutionText.textContent = '';
                loadingIndicator.classList.remove('hidden');
                contactFormContainer.classList.add('hidden'); // Hide form when generating new solution
                contactConfirmation.classList.add('hidden'); // Hide confirmation message
                generateSolutionBtn.disabled = true; // Disable button during loading

                try {
                    let chatHistory = [];
                    chatHistory.push({ role: "user", parts: [{ text: `As an AI expert for an AI agency named solidiA, suggest a concise and innovative AI solution (2-3 sentences) for the following business problem: "${prompt}". Focus on how AI can directly solve it and mention a relevant solidiA service.` }] });
                    const payload = { contents: chatHistory };
                    const apiKey = ""; // If you want to use models other than gemini-2.0-flash or imagen-3.0-generate-002, provide an API key here. Otherwise, leave this as-is.
                    const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                    const response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    const result = await response.json();

                    if (result.candidates && result.candidates.length > 0 &&
                        result.candidates[0].content && result.candidates[0].content.parts &&
                        result.candidates[0].content.parts.length > 0) {
                        const text = result.candidates[0].content.parts[0].text;
                        solutionText.textContent = text;
                        contactFormContainer.classList.remove('hidden'); // Show contact form after solution
                    } else {
                        solutionText.textContent = 'Could not generate a solution. Please try again.';
                        console.error('Gemini API response structure unexpected:', result);
                    }
                } catch (error) {
                    solutionText.textContent = 'An error occurred while generating the solution. Please try again later.';
                    console.error('Error calling Gemini API:', error);
                } finally {
                    loadingIndicator.classList.add('hidden');
                    generateSolutionBtn.disabled = false; // Re-enable button
                }
            });

            // Handle Contact Form Submission
            contactForm.addEventListener('submit', async (event) => { // Added async here
                event.preventDefault(); // Prevent default form submission

                const clientName = document.getElementById('clientName').value;
                const clientEmail = document.getElementById('clientEmail').value;
                const clientPhone = document.getElementById('clientPhone').value;

                const formData = {
                    name: clientName,
                    email: clientEmail,
                    phone: clientPhone,
                    solution_idea: solutionText.textContent // Include the generated solution idea
                };

                // Show loading indicator for form submission (optional, but good UX)
                contactForm.querySelector('button[type="submit"]').disabled = true;
                contactForm.querySelector('button[type="submit"]').textContent = 'Sending...';

                try {
                    // Send data to your VPS backend
                    const backendUrl = '/api/contact'; // This is the endpoint on your VPS
                    const response = await fetch(backendUrl, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify(formData),
                    });

                    if (response.ok) {
                        contactConfirmation.textContent = 'Thank you! Your request has been received. A solidiA expert will contact you soon.';
                        contactConfirmation.classList.remove('hidden');
                        contactConfirmation.classList.remove('bg-red-700'); // Remove error styling if present
                        contactConfirmation.classList.add('bg-blue-900'); // Add success styling
                        contactForm.reset(); // Clear the form
                        contactFormContainer.classList.add('hidden'); // Hide the form after successful submission
                    } else {
                        const errorData = await response.json();
                        contactConfirmation.textContent = `Error submitting form: ${errorData.message || 'Unknown error'}`;
                        contactConfirmation.classList.remove('hidden');
                        contactConfirmation.classList.remove('bg-blue-900'); // Remove success styling
                        contactConfirmation.classList.add('bg-red-700'); // Add error styling
                        console.error('Backend submission error:', errorData);
                    }
                } catch (error) {
                    contactConfirmation.textContent = 'An error occurred while sending your request. Please try again.';
                    contactConfirmation.classList.remove('hidden');
                    contactConfirmation.classList.remove('bg-blue-900');
                    contactConfirmation.classList.add('bg-red-700');
                    console.error('Network or other error during form submission:', error);
                } finally {
                    contactForm.querySelector('button[type="submit"]').disabled = false;
                    contactForm.querySelector('button[type="submit"]').textContent = 'Request a Call';
                }
            });
        });

            setCanvasSize();
            init(); // Re-initialize particles on resize
        });

                mouse.y = event.touches[0].clientY;
            }
        });

        window.addEventListener('mouseout', () => {
            mouse.x = null;
            mouse.y = null;
        });

        window.addEventListener('touchend', () => {
            mouse.x = null;
            mouse.y = null;
        });
